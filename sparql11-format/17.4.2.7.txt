#!/usr/bin/env spang2
# @option --fmt

PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX eg: <http://biometrics.example/ns#>

SELECT ?name ?shoeSize
WHERE {
  ?x foaf:name ?name .
  ?x eg:shoeSize ?shoeSize .
  FILTER (datatype(?shoeSize) = xsd:integer)
}
